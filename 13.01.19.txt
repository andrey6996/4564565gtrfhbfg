void print(float* m, int size)
{

}


void fill(float* m, int size)
{
	for (int i = 0; i < size; i++)
	{
		m[i] = (rand() % 100 - 50) / 10.0;
	}
}
float max_mas(float* m,int size)
{
	float max = m[0];
	for (int i = 0; i < size; i++)
	{
		if (m[i] > max)
		{
			max = m[i];
		}
	}
	return max;
}
